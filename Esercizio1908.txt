CREATE TABLE Clienti (
    NumeroCliente INT PRIMARY KEY,
    Nome VARCHAR(10),

    RagioneResidenza VARCHAR(20)

    IdProdotto INT PRIMARY KEY,
	Descrizione VARCHAR(200),
	InProduzione VARCHAR(20),
	InCommercio VARCHAR(20),
	DataAttivazione INT,
	DataDisattivazione INT,
